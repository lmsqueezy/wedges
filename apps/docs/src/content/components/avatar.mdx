### Usage

```tsx
import { Avatar } from "@lmsqueezy/wedges";
```
    
```tsx showLineNumbers
<Avatar alt='Image alt text' src="https://images.unsplash.com/photo-1517841905240-472988babdf9?q=80&w=250&h=250&auto=format&fit=crop" />
```

For more advanced usage, you can use the `Avatar.Root` component to compose your own avatar.

```tsx
<Avatar.Root>
    <Avatar.Image
        src="https://images.unsplash.com/photo-1517841905240-472988babdf9?q=80&w=250&h=250&auto=format&fit=crop&crop=face"
        alt="Max Quest"
    />
    <Avatar.Fallback>MQ</Avatar.Fallback>
    <Avatar.Notification />
    <Avatar.Status />
</Avatar.Root>
```
