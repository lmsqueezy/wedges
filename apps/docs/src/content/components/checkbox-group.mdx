### Usage

```tsx
import { CheckboxGroup } from "@lemonsqueezy/wedges";
```

```tsx showLineNumbers
<CheckboxGroup label="Group Label">
  <CheckboxGroup.Item label="Option 1" />
  <CheckboxGroup.Item label="Option 2" />
</CheckboxGroup>
```
